  function rewriteDeliveryAndPaymentWarnings() {
    var root = document.querySelector('.checkout-v2-wrapper, .co-section--checkout_content') || document.body;
    if (!root) return;

    var leaves = root.querySelectorAll('p, div, span, li');
    var localityEmpty = isLocalityEmpty();

    for (var i = 0; i < leaves.length; i++) {
      var node = leaves[i];
      if (!node || node.children.length) continue;

      var text = (node.textContent || '').trim();
      if (!text || text.length > 180) continue;

      if (!localityEmpty) {
        if (node.dataset && node.dataset.originalStatusText) {
          node.textContent = node.dataset.originalStatusText;
          delete node.dataset.originalStatusText;
        }
        continue;
      }

      if (/доставк[аеи]?\s+для\s+пункт[ауы]\s+недоступн/i.test(text)) {
        if (node.dataset && !node.dataset.originalStatusText) {
          node.dataset.originalStatusText = text;
        }
        node.textContent = EMPTY_LOCALITY_DELIVERY_TEXT;
        continue;
      }

      if (/недоступн[а-я]*\s+оплат|оплат[а-я]*\s+недоступн/i.test(text)) {
        if (node.dataset && !node.dataset.originalStatusText) {
          node.dataset.originalStatusText = text;
        }
        node.textContent = EMPTY_LOCALITY_PAYMENT_TEXT;
      }
    }
  }

  function getSelectedDeliveryType() {
    var radios = document.querySelectorAll('input[name="order[delivery_variant_id]"]');
    for (var i = 0; i < radios.length; i++) {
      if (radios[i].checked) {
        var label = document.querySelector('label[for="' + radios[i].id + '"]');
        if (!label) continue;
        var labelText = label.textContent || '';
        var isCourier = /курьер/i.test(labelText) || /до\s+двери/i.test(labelText);
        var isPVZ = /пункт/i.test(labelText) || /выдач/i.test(labelText) || /автолайт/i.test(labelText);
        return isCourier ? 'courier' : (isPVZ ? 'pvz' : 'other');
      }
    }
    return null;
  }

  function getTimeIntervalWrapper() {

  function getSelectedDeliveryZone() {
    // Зона зашита в tariff_id формата "courier_door_saturday|district|village"
    var radios = document.querySelectorAll('input[name="order[delivery_variant_id]"]');
    for (var i = 0; i < radios.length; i++) {
      if (!radios[i].checked) continue;
      var label = document.querySelector('label[for="' + radios[i].id + '"]');
      var container = (label && label.closest('.co-delivery-variant, .co-radio, .co-checkout-block')) ||
                      radios[i].closest('.co-delivery-variant, .co-radio, .co-checkout-block') ||
                      label || radios[i].parentNode;
      if (!container) continue;
      var html = container.outerHTML || '';
      var m = html.match(/courier_door_(saturday|district|village)/i);
      if (m) return m[1].toLowerCase();
      return null;
    }
    return null;
  }

  function isCourierDeliverySelected() {
    var radios = document.querySelectorAll('input[name="order[delivery_variant_id]"]');
    for (var i = 0; i < radios.length; i++) {
      if (!radios[i].checked) continue;
      var label = document.querySelector('label[for="' + radios[i].id + '"]');
      var text = label ? (label.textContent || '') : '';
      var html = label ? (label.outerHTML || '') : '';
      if (/курьер/i.test(text) || /до\s+двери/i.test(text)) return true;
      if (/courier_door_/i.test(html)) return true;
      return false;
    }
    return false;
  }

  function updateTimeIntervalVisibility() {
    var wrapper = getTimeIntervalWrapper();
    if (!wrapper) return;

    var courier = isCourierDeliverySelected();
    var zone = getSelectedDeliveryZone();
    var shouldShow = courier && (zone === 'saturday');

    if (shouldShow) {
      wrapper.style.display = '';
    } else {
      wrapper.style.display = 'none';
      // Сбрасываем значение, чтобы оно не уехало в заказ
      var sel = getTimeIntervalSelect();
      if (sel) sel.value = '';
    }
  }

  function attachTimeIntervalListeners() {
    var radios = document.querySelectorAll('input[name="order[delivery_variant_id]"]');
    radios.forEach(function(radio) {
      if (!radio.dataset.intervalListenerAttached) {
        radio.addEventListener('change', updateTimeIntervalVisibility);
        radio.dataset.intervalListenerAttached = '1';
      }
    });

    var cityField = document.getElementById('shipping_address_full_locality_name');
    if (cityField && !cityField.dataset.intervalListenerAttached) {
      cityField.addEventListener('change', function() {
        // Даём InSales время пересчитать варианты доставки после смены города
        setTimeout(updateTimeIntervalVisibility, 300);
      });
      cityField.dataset.intervalListenerAttached = '1';
    }
  }

  function attachAddressVisibilityListeners() {
    var cityField = document.getElementById('shipping_address_full_locality_name');
    var addressField = document.getElementById('shipping_address_address');
    var radios = document.querySelectorAll('input[name="order[delivery_variant_id]"]');
    
    if (cityField && !cityField.dataset.addressListenerAttached) {
      cityField.addEventListener('input', function() {
        updateAddressFieldVisibility();
        updateWarningsVisibility();
      });
      cityField.addEventListener('change', function() {
        updateAddressFieldVisibility();
        updateWarningsVisibility();
      });
      cityField.dataset.addressListenerAttached = '1';
    }
    
    if (addressField && !addressField.dataset.addressListenerAttached) {
      addressField.addEventListener('input', function() {
        updateAddressFieldVisibility();
        checkAddressForSettlement();
      });
      addressField.addEventListener('change', function() {
        updateAddressFieldVisibility();
        checkAddressForSettlement();
      });
      addressField.dataset.addressListenerAttached = '1';
    }
    
    radios.forEach(function(radio) {
      if (!radio.dataset.addressListenerAttached) {
        radio.addEventListener('change', updateAddressFieldVisibility);
        radio.dataset.addressListenerAttached = '1';
      }
    });
  }

    var desc = loginBlock.querySelector('.co-input-description');
    if (desc) desc.style.display = 'none';
    if (btn) btn.style.display = 'none';
    if (title && !loginBlock.querySelector('#' + LOGIN_LINK_ID)) {
      title.textContent = 'Уже делали заказ у нас?';
      var link = document.createElement('a');
      link.id = LOGIN_LINK_ID;
      link.href = '#';
      link.textContent = 'Войти в личный кабинет';
      link.addEventListener('click', function(e) {
        e.preventDefault();
        if (btn) btn.click();
      });
      title.parentNode.insertBefore(link, title.nextSibling);
    }
    if (!document.getElementById(LOGIN_STYLE_ID)) {
      var style = document.createElement('style');
      style.id = LOGIN_STYLE_ID;
      style.textContent = [
        '.co-login-block .co-title--h2{font-size:18px;font-weight:600;color:#2b2b2b;margin:0 0 6px 0;}',
        '#' + LOGIN_LINK_ID + '{display:inline-block;margin:0 0 8px 0;color:#005844;text-decoration:underline;font-size:16px;font-weight:600;}',
        '#' + LOGIN_LINK_ID + ':hover{color:#0b6a53;}',
        '.co-login-block .co-button--checkout-login{display:none !important;}'
      ].join('');
      document.head.appendChild(style);
    }
  }
  
  // === Перемещение блока авторизации ===
  function moveLoginBlock() {
    var loginBlock = document.querySelector('.co-login-block');
    if (!loginBlock) return false;
    var container = loginBlock.closest('.co-section--checkout_customer') || loginBlock.parentNode;
    if (!container) return false;
    var contactTitle = null;
    var titles = container.querySelectorAll('.co-title--h2, .co-title--h3');
    for (var i = 0; i < titles.length; i++) {
      if (/^\s*Контактные данные\s*$/i.test(titles[i].textContent || '')) {
        contactTitle = titles[i];
        break;
      }
    }
    var contactBlock = null;
    if (contactTitle) {
      contactBlock = contactTitle.closest('.co-checkout-block') || contactTitle.parentNode;
    } else {
      var contactFields = container.querySelector('.co-contact-fields');
      contactBlock = contactFields ? (contactFields.closest('.co-checkout-block') || contactFields.parentNode) : null;
    }
    if (!contactBlock || !contactBlock.parentNode) return false;
    if (contactBlock.nextSibling === loginBlock) return true;
    contactBlock.parentNode.insertBefore(loginBlock, contactBlock.nextSibling);
    return true;
  }
  
  function renameSubmitButton() {
    var btn = document.querySelector('.co-button--checkout, .co-button--checkout_submit, [data-checkout-submit], button[type="submit"]');
    if (!btn || btn.dataset.renamed) return;
    if (btn.textContent && btn.textContent.trim()) {
      btn.textContent = 'Подтвердить и оформить заказ';
      btn.dataset.renamed = '1';
    }
  }
  
  // === Фикс: InSales не снимает disabled с кнопки после пересчёта доставки ===
  // При смене города InSales ставит disabled на кнопку (AJAX-запрос к API доставки),
  // но после получения ответа не снимает его — кнопка остаётся заблокированной.
  // Решение: наблюдаем за #delivery_variants — InSales вставляет в него новые варианты
  // доставки сразу после завершения пересчёта, и тогда снимаем disabled с кнопки.
  var _submitBtnUnlockTimer = null;
  function watchDeliveryPriceUpdate() {
    var variantsEl = document.getElementById('delivery_variants');
    if (!variantsEl || variantsEl.dataset.btnWatcherAttached) return;
    variantsEl.dataset.btnWatcherAttached = '1';

    var obs = new MutationObserver(function(muts) {
      var hasChildListChange = muts.some(function(m) { return m.type === 'childList' && m.addedNodes.length > 0; });
      if (!hasChildListChange) return;
      clearTimeout(_submitBtnUnlockTimer);
      _submitBtnUnlockTimer = setTimeout(function() {
        var btn = document.getElementById('create_order') ||
                  document.querySelector('.js-button-checkout_submit, .co-button--checkout_submit');
        if (btn && btn.disabled) {
          btn.disabled = false;
        }
      }, 200);
    });
    obs.observe(variantsEl, { childList: true, subtree: false });
  }

  var loginMoved = false;
  var _runAllBusy = false;
  function runAll() {
    if (_runAllBusy) return;
    _runAllBusy = true;
    try {
      insertAllHints();
      insertNameHint();
      insertEmailHint();
      insertDeliveryHints();
      ensureSubmitButtonStyle();
      ensureCheckoutTitleStyle();
      renameSubmitButton();
      ensureSubmitSmsNotice();
      updateWarningsVisibility();
      if (typeof setupPhoneField === 'function') {
        setupPhoneField();
      }
      pimpLoginBlock();
      if (!loginMoved) {
        loginMoved = moveLoginBlock();
      }
      attachAddressVisibilityListeners();
      updateAddressFieldVisibility();
      checkAddressForSettlement();
      attachTimeIntervalListeners();
      updateTimeIntervalVisibility();
      watchDeliveryPriceUpdate();
    } catch (e) {
      // Защита от ошибок, чтобы не сломать процесс оформления
    }
    setTimeout(function() { _runAllBusy = false; }, 50);
  }
  
  function startObserver() {
    if (!document.body) return;
    runAll();
    var observer = new MutationObserver(runAll);
    observer.observe(document.body, { childList: true, subtree: true });
  }
  
  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', startObserver);
  } else {
    startObserver();
  }
})();
